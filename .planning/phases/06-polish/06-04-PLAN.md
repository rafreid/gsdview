# Plan 06-04: Recent projects storage and UI

## Goal
Store recently opened projects and show them for quick access.

## Tasks

1. Use electron-store or localStorage for persistence
2. Store up to 5 recent project paths
3. Add recent projects dropdown to toolbar
4. Click to load recent project

## Implementation

### Main process - store recent projects:
```javascript
const Store = require('electron-store');
const store = new Store();

ipcMain.handle('get-recent-projects', () => {
  return store.get('recentProjects', []);
});

ipcMain.handle('add-recent-project', (event, projectPath) => {
  let recent = store.get('recentProjects', []);
  recent = recent.filter(p => p !== projectPath);
  recent.unshift(projectPath);
  recent = recent.slice(0, 5);
  store.set('recentProjects', recent);
  return recent;
});
```

### Preload:
```javascript
getRecentProjects: () => ipcRenderer.invoke('get-recent-projects'),
addRecentProject: (path) => ipcRenderer.invoke('add-recent-project', path)
```

### Renderer - dropdown:
```javascript
async function updateRecentProjects() {
  if (!window.electronAPI.getRecentProjects) return;
  const recent = await window.electronAPI.getRecentProjects();
  const dropdown = document.getElementById('recent-projects');
  dropdown.innerHTML = '<option value="">Recent Projects...</option>';
  for (const path of recent) {
    const opt = document.createElement('option');
    opt.value = path;
    opt.textContent = path.split('/').pop();
    dropdown.appendChild(opt);
  }
}

// In loadProject, after success:
if (window.electronAPI.addRecentProject) {
  await window.electronAPI.addRecentProject(projectPath);
  updateRecentProjects();
}
```
